Integration

copy · paste · ship
Plain HTML
<!-- include once -->
<script src="pip-mascot.js"></script>

<!-- use anywhere -->
<pip-mascot pose="wave" size="160"></pip-mascot>
<pip-mascot pose="celebrate"></pip-mascot>
<pip-mascot autocycle="3000"></pip-mascot>
React
// index.html: include the script once
// component:
export function Mascot({ pose }) {
  return <pip-mascot pose={pose}></pip-mascot>;
}
Imperative API
const el = document.querySelector('pip-mascot');

// change pose
el.pose = 'celebrate';

// cycle through some poses
el.cycle(['wave', 'think', 'point'], 2500);
el.stopCycle();

// listen for changes
el.addEventListener('pose-change', e => {
  console.log(e.detail.pose);
});
Attributes
pose       wave | think | point | celebrate | sleep | oops
size       number (px) or any CSS length
speed      0.3–3 — animation speed multiplier
autocycle  ms between auto pose changes
no-shadow  hide ground shadow ellipse
